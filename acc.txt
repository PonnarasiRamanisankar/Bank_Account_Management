    Account_Number VARCHAR2(10) PRIMARY KEY,
    Customer_Name VARCHAR2(15),
    Balance NUMBER(10,2)
);

CREATE TABLE TRANSFER_TBL (
    Transaction_ID NUMBER(4) PRIMARY KEY,
    Account_Number VARCHAR2(10) REFERENCES ACCOUNT_TBL(Account_Number),
    Beneficiary_Account_Number VARCHAR2(10) REFERENCES ACCOUNT_TBL(Account_Number),
    Transaction_Date DATE,
    Transaction_Amount NUMBER(10,2),
    Account_To_Be_Transferred VARCHAR2(10) REFERENCES ACCOUNT_TBL(Account_Number)
);

CREATE SEQUENCE TRANSACTIONID_SEQ
START WITH 1000
INCREMENT BY 1
MINVALUE 1000
MAXVALUE 9999
NOCACHE
NOCYCLE;

SELECT TRANSACTIONID_SEQ.NEXTVAL FROM dual;
